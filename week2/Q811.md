# 811. 子域名访问计数
> https://leetcode-cn.com/problems/subdomain-visit-count/

```python
class Solution:
    def subdomainVisits(self, cpdomains: List[str]) -> List[str]:
        dic = {}
        res = []
        for cpdomain in cpdomains:
            temp1 = cpdomain.split()
            count = int(temp1[0])
            temp2 = temp1[1].split('.')
            if len(temp2) == 3:
                dic[temp2[2]] = dic.get(temp2[2], 0) + count
                dic[temp2[1]+'.'+temp2[2]] = dic.get(temp2[1]+'.'+temp2[2], 0) + count
                dic[temp2[0]+'.'+temp2[1]+'.'+temp2[2]] = dic.get(temp2[0]+'.'+temp2[1]+'.'+temp2[2], 0) + count
            elif len(temp2) == 2:
                dic[temp2[1]] = dic.get(temp2[1], 0) + count
                dic[temp2[0]+'.'+temp2[1]] = dic.get(temp2[0]+'.'+temp2[1], 0) + count
        
        for key in dic.keys():
            res.append(str(dic[key])+' '+key)
        return res
```
